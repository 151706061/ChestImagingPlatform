CMAKE_MINIMUM_REQUIRED(VERSION 2.6)

PROJECT(RegisterLabelMaps)

set(MODULE_NAME RegisterLabelMaps)

find_package(LibXml2 REQUIRED)

find_package(SlicerExecutionModel REQUIRED)
include(${SlicerExecutionModel_USE_FILE})


 FIND_PACKAGE( ITK REQUIRED )

 FIND_PACKAGE( CIP REQUIRED )

 INCLUDE( ${CIP_USE_FILE} )

set(MODULE_INCLUDE_DIRECTORIES
  ${CIP_INCLUDE_DIRECTORIES}
  ${LIBXML2_INCLUDE_DIR}
  )

set(MODULE_TARGET_LIBRARIES
  ${ITK_LIBRARIES}
  ${CIP_LIBRARIES}
  ${LIBXML2_LIBRARIES}
  )

set(MODULE_SRCS
  )

SEMMacroBuildCLI(
  NAME ${MODULE_NAME}
  TARGET_LIBRARIES ${MODULE_TARGET_LIBRARIES}
  INCLUDE_DIRECTORIES ${MODULE_INCLUDE_DIRECTORIES}
  ADDITIONAL_SRCS ${MODULE_SRCS}
  RUNTIME_OUTPUT_DIRECTORY        "${CMAKE_BINARY_DIR}"
  ARCHIVE_OUTPUT_DIRECTORY        "${CMAKE_BINARY_DIR}"    
  INSTALL_LIBRARY_DESTINATION     ${CMAKE_BINARY_DIR}
  INSTALL_ARCHIVE_DESTINATION     ${CMAKE_BINARY_DIR}
  INSTALL_RUNTIME_DESTINATION     ${CMAKE_BINARY_DIR}
  EXECUTABLE_ONLY
  )
