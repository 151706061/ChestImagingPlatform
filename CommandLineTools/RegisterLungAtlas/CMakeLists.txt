CMAKE_MINIMUM_REQUIRED(VERSION 2.6)

PROJECT( RegisterLungAtlas )

set(MODULE_NAME RegisterLungAtlas)

find_package(SlicerExecutionModel REQUIRED)
include(${SlicerExecutionModel_USE_FILE})
 FIND_PACKAGE( ITK REQUIRED )

FIND_PACKAGE ( VTK )
IF ( VTK_FOUND )
  INCLUDE(${VTK_USE_FILE})
ELSE ( VTK_FOUND )
  MESSAGE ( FATAL_ERROR "Cannot build without VTK" )
ENDIF ( VTK_FOUND )


FIND_PACKAGE( CIP REQUIRED )


set(MODULE_INCLUDE_DIRECTORIES
  ${CIP_INCLUDE_DIRECTORIES}
  )

set(MODULE_TARGET_LIBRARIES
  ${ITK_LIBRARIES}
  ${CIP_LIBRARIES}
  ${VTK_LIBRARIES}
  )

set(MODULE_SRCS
  )

SET ( VTK_LIBS
vtkIO 
vtkCommon 
vtkImaging 
vtkFiltering 
vtkGraphics 
vtkRendering 
vtkWidgets 
vtkHybrid
)

SEMMacroBuildCLI(
  NAME ${MODULE_NAME}
  TARGET_LIBRARIES ${MODULE_TARGET_LIBRARIES}
  INCLUDE_DIRECTORIES ${MODULE_INCLUDE_DIRECTORIES}
  ADDITIONAL_SRCS ${MODULE_SRCS}
  RUNTIME_OUTPUT_DIRECTORY        "${CMAKE_BINARY_DIR}"
  ARCHIVE_OUTPUT_DIRECTORY        "${CMAKE_BINARY_DIR}"    INSTALL_LIBRARY_DESTINATION     ${CMAKE_BINARY_DIR}
  INSTALL_ARCHIVE_DESTINATION     ${CMAKE_BINARY_DIR}
  INSTALL_RUNTIME_DESTINATION     ${CMAKE_BINARY_DIR}
  EXECUTABLE_ONLY
  )



INCLUDE_DIRECTORIES( ${CMAKE_SOURCE_DIR}/Common )
INCLUDE_DIRECTORIES( ${CMAKE_SOURCE_DIR}/Utilities )

SET ( RegisterLungAtlas_SOURCE RegisterLungAtlas.cxx )

#ADD_EXECUTABLE ( RegisterLungAtlas ${RegisterLungAtlas_SOURCE} )
TARGET_LINK_LIBRARIES ( RegisterLungAtlas ${VTK_LIBRARIES} ${ITK_LIBRARIES} )
